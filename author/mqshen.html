vimrc配置:

"启用filetype插件
filetype plugin on
filetype indent on
syntax on

"fast saving
nmap <leader>w :w!<cr>

"fast editing of the .vimrc
map <leader>e :e! ~/.vim_runtime/vimrc<cr>
