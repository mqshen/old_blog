    if version >= 703
        setl foldmethod=syntax
        setl foldlevelstart=1
    endif

    if version >= 703

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""  
" 语言的编译和运行             
" 支持的语言：java         F5编译(保存+编译)  F6运行  
"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""  
func! CompileCode()  
    exec "w"  
    if &filetype == "java"  
        exec "!javac -encoding utf-8 %"  
    endif  
endfunc  
func! RunCode()  
    if &filetype == "java"  
        exec "!java -classpath %:h; %:t:r"  
    endif  
endfunc

" F6 保存+编译  
map <F6> :call CompileCode()<CR>

"  F7 运行  
map <F7> :call RunCode()<CR>

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""

"--------引号 && 括号自动匹配  
:inoremap ( ()<ESC>i  
:inoremap ) <c-r>=ClosePair(')')<CR>  
:inoremap { {}<ESC>i  
:inoremap } <c-r>=ClosePair('}')<CR>  
:inoremap [ []<ESC>i  
:inoremap ] <c-r>=ClosePair(']')<CR>  
:inoremap < <><ESC>i  
:inoremap > <c-r>=ClosePair('>')<CR>  
:inoremap " ""<esc>i  
:inoremap ' ''<ESC>i  
:inoremap ` ``<ESC>i  
function ClosePair(char)  
    if getline('.')[col('.') - 1] == a:char  
        return "\<Right>"  
    else  
        return a:char  
    endif  
endf

#!/bin/sh
CURRENT_YEAR=`date +'%Y'`
CURRENT_MONTH=`date +'%m'`
CURRENT_DAY=`date +'%d'`
CURRENT_DATE=`date +'%Y-%m-%d'`
POST_DIRECTORY='posts'

if [ "$1" == "newPost" ]; then
    if test ! -d $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH; then
        echo 'create directory '$POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH
        mkdir -p $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH
    fi
    if test -f $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH/$2.md; then
        echo $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH/$2.md 'is exists'
        exit 1
    fi 
    echo 'create file '$POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH/$2.md
    echo 'Date: '$CURRENT_DATE >> $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH/$2.md
    echo 'Title: '$2 >> $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH/$2.md
    echo 'Category: ' >> $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH/$2.md
    echo 'Tags: ' >> $POST_DIRECTORY/$CURRENT_YEAR/$CURRENT_MONTH/$2.md
    exit 0
fi

if [ "$1" == "deploy" ]; then
    pelican ./ -t ./ -s settings.py 
    cd output
    git add .
    git commit -m $2
    git push
    exit 0
fi
